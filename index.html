function processKinshipTopologyTree() { const treeSection = document.getElementById('familyTreeSection'); const treeContainer = document.getElementById('familyTreeContainer'); treeContainer.innerHTML = ''; if (queryCapturedRecords.length === 0) { treeSection.style.display = 'none'; return; } let fatherGroups = {}; let addressGroups = {}; let uniqueProfiles = []; let seenProfileKeys = new Set(); // Helper to extract a highly standardized phonetic/canonical base key function getCanonicalKey(str) { if (!str || str === 'N/A') return ''; return str.toUpperCase() .replace(/S\/O\s+/g, '') // Cleans out father prefix notations cleanly .replace(/D\/O\s+/g, '') .replace(/W/O\s+/g, '') .replace(/[^A-Z0-9]/g, '') // Strips commas, periods, spaces, and punctuation blocks .replace(/KAMATA/g, 'KAMTA') // Explicit phonological consolidation mapping edge-case fix .trim(); } // Deduplicate incoming raw data loops securely queryCapturedRecords.forEach(rec => { let nameClean = (rec.name || '').trim().replace(/\s+/g, ' '); let fatherClean = (rec.father_name || '').trim().replace(/\s+/g, ' '); let uniqueKey = `${nameClean.toUpperCase()}||${fatherClean.toUpperCase()}`; if (!seenProfileKeys.has(uniqueKey) && nameClean) { seenProfileKeys.add(uniqueKey); uniqueProfiles.push({ name: nameClean, father_name: fatherClean, mobile: rec.mobile || 'N/A', address: (rec.address || '').trim().replace(/\s+/g, ' ') }); } }); // Cluster matching elements across entities securely uniqueProfiles.forEach(p => { let rawFatherKey = getCanonicalKey(p.father_name); let rawAddrKey = getCanonicalKey(p.address); if (rawFatherKey && rawFatherKey.length > 2 && rawFatherKey !== 'NOTDECLARED') { if (!fatherGroups[rawFatherKey]) { fatherGroups[rawFatherKey] = { displayParentName: p.father_name.replace(/S\/O\s+/gi, '').toUpperCase(), members: [] }; } // Enforces that we only store a single individual once per parent index group cluster block if (!fatherGroups[rawFatherKey].members.some(m => m.name.toUpperCase() === p.name.toUpperCase())) { fatherGroups[rawFatherKey].members.push(p); } } if (rawAddrKey && rawAddrKey.length > 10) { // Isolate a stable geographic threshold chunk string (20 anchor vectors) let baseGeographicToken = rawAddrKey.substring(0, 20); if (!addressGroups[baseGeographicToken]) { addressGroups[baseGeographicToken] = { displayAddressString: p.address, residents: [] }; } if (!addressGroups[baseGeographicToken].residents.some(r => r.name.toUpperCase() === p.name.toUpperCase())) { addressGroups[baseGeographicToken].residents.push(p); } } }); let outputText = ""; // Render Section A: Multi-Individual Sibling Branches Manifest let siblingSectionsFound = false; for (let fKey in fatherGroups) { let group = fatherGroups[fKey]; if (group.members.length > 1) { siblingSectionsFound = true; outputText += `[+] PARENT ROOT SEED: ${group.displayParentName}\n`; outputText += ` │\n`; outputText += ` ├── STATUS: CRITICAL RELATION LINK RESOLVED -> [SIBLINGS DETECTED]\n`; group.members.forEach((profile, index) => { let pointer = (index === group.members.length - 1) ? '└──' : '├──'; outputText += ` ${pointer} MEMBER Node: ${profile.name.toUpperCase()} (Mob: ${profile.mobile})\n`; }); outputText += `\n`; } } // Render Section B: Co-Located Residency Cells Manifest let locationSectionsFound = false; for (let aKey in addressGroups) { let group = addressGroups[aKey]; if (group.residents.length > 1) { locationSectionsFound = true; let clippedDisplayAddr = group.displayAddressString; if(clippedDisplayAddr.length > 50) clippedDisplayAddr = clippedDisplayAddr.substring(0, 50) + "..."; outputText += `[λ] GEOGRAPHIC AREA VECTOR: ${clippedDisplayAddr.toUpperCase()}\n`; outputText += ` │\n`; outputText += ` ├── STATUS: DOMESTIC CLUSTER BOUND -> [CO-LOCATED AT RESIDENCY UNIT]\n`; group.residents.forEach((profile, index) => { let pointer = (index === group.residents.length - 1) ? '└──' : '├──'; outputText += ` ${pointer} TARGET Node: ${profile.name.toUpperCase()} [Father: ${profile.father_name}]\n`; }); outputText += `\n`; } } if (!siblingSectionsFound && !locationSectionsFound) { outputText = `> NO RELATIONAL KINSHIP CLUSTERS DETECTED ACROSS LOG PAYLOADS.\n> Target identities represent separate independent profiles.`; } treeSection.style.display = 'block'; treeContainer.innerText = outputText; }
